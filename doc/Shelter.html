def self.order_by_number_of_pets
  select("shelters.*, count(pets.id) AS pets_count")
    .joins("LEFT OUTER JOIN pets ON pets.shelter_id = shelters.id")
    .group("shelters.id")
    .order("pets_count DESC")
end
